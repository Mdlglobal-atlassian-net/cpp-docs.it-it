# <a name="compiler-error-c3724"></a>Errore del compilatore C3724
è necessario #include \<Windows. h > utilizzare multithreading con gli eventi  
  
 Il file di Windows. h è obbligatorio se si usa il multithreading con gli eventi. Per correggere l'errore, aggiungere `#include <windows.h>` all'inizio del file in cui le origini eventi e l'evento riceventi.  
  
```  
// C3724.cpp  
// uncomment the following line to resolve  
// #include <windows.h>  
  
[event_source(native), threading(apartment)]  
class CEventSrc {  
public:  
   __event void event1();   // C3724  
};  
  
[event_receiver(native)]  
class CEventRec {  
public:  
   void handler1() {  
   }  
  
   void HookEvents(CEventSrc* pSrc) {  
      __hook(CEventSrc::event1, pSrc, CEventRec::handler1);  
   }  
  
   void UnhookEvents(CEventSrc* pSrc) {  
      __unhook(CEventSrc::event1, pSrc, CEventRec::handler1);  
   }  
};  
  
int main() {  
}  
```
