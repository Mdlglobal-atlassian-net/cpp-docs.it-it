# Errore del compilatore C2296
[!INCLUDE[vs2017banner](../../assembler/inline/includes/vs2017banner.md)]

l'operando sinistro ha il tipo 'operatore'. Non valido  
  
 L'operando sinistro utilizzato con `operator` non è valido.  
  
 È ad esempio possibile che il compilatore rilevi una dichiarazione anziché la chiamata di funzione prevista.  
  
 Il seguente codice di esempio genera l'errore C2296:  
  
```  
// C2296.cpp  
struct MyStruct {  
   struct Help {  
      Help(float f) : m_f(f) {}  
      float m_f;  
   };  
  
   MyStruct(const Help &h) : m_f(h.m_f) {}  
   MyStruct(float f) : m_f(f) {}  
   MyStruct operator*(const MyStruct &f1) const {   
      return MyStruct(m_f * f1.m_f);  
   }  
  
private:  
   float m_f;  
};  
  
int main() {  
   float f1 = 1.0f;  
  
   MyStruct m_MyStruct1 ( MyStruct::Help( f1 ) );  
   // try the following line instead  
   // MyStruct m_MyStruct1 = MyStruct::Help( f1 );  
  
   MyStruct m_MyStruct2 ( MyStruct::Help( f1 ) );  
   // try the following line instead  
   // MyStruct m_MyStruct2 = MyStruct::Help( f1 );  
  
   MyStruct m_MyStruct3 = m_MyStruct1 * m_MyStruct2;   // C2296  
}  
```
